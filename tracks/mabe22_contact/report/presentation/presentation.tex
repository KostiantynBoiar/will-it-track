% ------------------------------------------------------------ the question
\begin{frame}{The question in one line}
\vspace{0.6em}
\begin{block}{}
\centering\large
    A \textbf{social behaviour} is defined by \textbf{how two bodies meet} —
    how close, how much they overlap, where they touch.
\end{block}
\vspace{0.8em}
\begin{itemize}
\item The field reads that geometry from \textbf{keypoint hulls}.
\item \textbf{We ask:} can a \textbf{segmentation mask} read it better —
          exactly where hulls are weakest, \emph{during contact}?
\end{itemize}
\end{frame}

% ----------------------------------------------- the weakness of hulls
\begin{frame}{Why hulls fail \emph{during} contact}
\begin{columns}[T]
\begin{column}{0.54\textwidth}
\vspace{1.2em}
\begin{itemize}\itemsep0.7em
\item Keypoints \textbf{occlude} one another.
\item On look-alike mice the tracker \textbf{swaps} them.
\item The polygon \textbf{smooths over the true boundary}.
\end{itemize}
\vspace{0.6em}
\begin{block}{The crux}
        Crudest at the very moment the behaviour is \emph{defined}.
\end{block}
\end{column}
\begin{column}{0.44\textwidth}
\centering
\begin{tikzpicture}[scale=0.8]

% ----------------------------------------------------- experiment / ablations
\begin{frame}{The experiment (pre-registered)}
\begin{enumerate}\itemsep0.6em
\item[\textbf{a}] \textbf{Hull geometry} — the baseline.
\item[\textbf{b}] \textbf{Mask geometry} vs (a) — \textcolor{burnt}{\textbf{the head-to-head}}.
\item[\textbf{c}] \textbf{Fusion} on frozen VideoMAE — does mask geometry add anything extra?
\end{enumerate}
\vspace{0.5em}
\begin{itemize}\itemsep0.4em
\item Identical head; $\geq 5$ seeds; mean $\pm$ 95\,\% CI.
\item \textbf{Success fixed up front}; scrambled-mask control rules out a capacity artefact.
\end{itemize}
\vspace{0.3em}
\begin{block}{}
\centering Interpretable \textbf{either way} — a clean negative is still a result.
\end{block}
\end{frame}
% ------------------------------------------------------------- the one risk
\begin{frame}{The one make-or-break risk (settled in week 1)}
\centering\small
\emph{Can SAM\,2 keep three look-alike mice separate \textbf{through the contact frames}?}
\vspace{0.6em}
\begin{columns}[T]
\begin{column}{0.32\textwidth}
\begin{block}{\small PASS}
\footnotesize identities hold $\rightarrow$ proceed; contact geometry is the headline.
\end{block}
\end{column}
\begin{column}{0.34\textwidth}
\begin{block}{\small PARTIAL}
\footnotesize masks merge only in deepest contact $\rightarrow$ use the \emph{merge itself} as a
        high-contact feature.
\end{block}
\end{column}
\begin{column}{0.32\textwidth}
\begin{block}{\small FAIL}
\footnotesize frequent swaps $\rightarrow$ pivot to \textbf{background-robustness}; report a clean
        negative.
\end{block}
\end{column}
\end{columns}
\vspace{0.6em}
  A \textbf{week-1 pilot} on huddle/chase clips fires this decision. A clean spine either way.
\end{frame}

% ------------------------------------------------------------- takeaway
\begin{frame}{Contribution}
\begin{block}{}
\centering To our knowledge, the \textbf{first use of inter-animal segmentation-mask contact
    geometry} (mask IoU, contact-boundary length) as an explicit feature for social behaviour —
    in a clean head-to-head against the keypoint-hull geometry the field already uses.
\end{block}
\vspace{0.5em}
\begin{itemize}\itemsep0.4em
\item Honest: silhouette shape is \emph{not} new; the \textbf{mask-contact measurement} and the
\textbf{controlled comparison} are.
\item Every heavy model frozen; reproducible from cache; multi-seed CIs.
\item Built-in insurance: background-robustness if the masking pilot fails.
\end{itemize}
